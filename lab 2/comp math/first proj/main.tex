    public void backwardSubstitution() {
        for (int i = size - 1; i >= 0; i--) {
            double sum = 0;
            for (int j = i + 1; j < size; j++) {
                sum += matrix_a[i][j] * solutions[j];
            }
            solutions[i] = (matrix_b[i] - sum) / matrix_a[i][i];
        }
    }
